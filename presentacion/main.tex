\begin{frame}{Arquitectura del MLP (Clase 11, Clase 13 - Backprop)}
  \textbf{Forward}: $V_j^m = \theta\!\left(\sum_k V_k^{m-1} \cdot w_{jk}^m\right)$
  con $V^0 = x$, $O = V^M$.

  \textbf{Loss (MSE)}: $E(w) = \frac{1}{2N} \sum_{\mu=1}^N \sum_i \big(\zeta_i^\mu - O_i^\mu\big)^2$

  \textbf{Backward (regla de la cadena)}:
  \[
    \delta^{(L)} = (O - t) \odot f'(h^{(L)}), \quad
    \delta^{(m)} = \big(W^{(m+1)\top}\delta^{(m+1)}\big) \odot f'(h^{(m)})
  \]
  \[
    \frac{\partial E}{\partial W^{(m)}} = \frac{1}{N}\,\delta^{(m)}\,V^{(m-1)\top}, \quad
    w^{\text{nuevo}} = w^{\text{ant}} - \eta\,\frac{\partial E}{\partial w}
  \]

  \textbf{Capacidad y generalizaciĂ³n (Clase 13 - RegularizaciĂ³n)}:
  mĂ¡s capacidad $\Rightarrow$ mĂ¡s riesgo de overfitting; brecha
  train$-$validaciĂ³n crece con la capacidad.
\end{frame}

\begin{frame}{EJ1 --- MĂ©tricas de evaluaciĂ³n}
  \small
  Como criterio de diseĂ±o, queremos \textbf{minimizar los falsos negativos (FN)}:
  casos donde una operaciĂ³n fraudulenta se clasifica como no fraudulenta.
  Aun asĂ­, tambiĂ©n contemplamos el resto de los errores.

  \vspace{0.4em}
  \begin{itemize}
    \item \textbf{F2}
    \[
      F_\beta = (1 + \beta^2) \cdot \frac{\mathrm{Precision}\,\mathrm{Recall}}{\beta^2\,\mathrm{Precision} + \mathrm{Recall}}
      \qquad \Rightarrow \qquad
      F_2 = 5 \cdot \frac{\mathrm{Precision}\,\mathrm{Recall}}{4\,\mathrm{Precision} + \mathrm{Recall}}
    \]

    \item \textbf{MSE}
    \[
      \mathrm{MSE} = \frac{1}{N} \sum_{i=1}^{N} \left(y_i - \hat{y}_i\right)^2
    \]
    \scriptsize donde $y_i$ es la salida de \textbf{BigModel} y $\hat{y}_i$ es la predicciĂ³n de nuestro modelo.

    \item \textbf{Costo = $2\,FN + FP$}
  \end{itemize}
\end{frame}

\begin{frame}{Exp2 â€” Â¿Por quĂ© Adam con lr $= 10^{-3}$?}
  \begin{itemize}

    \item \textbf{GD} no tiene adaptaciĂ³n: necesita lr alto para

    \item \textbf{Momentum} queda en un punto intermedio: acelera

    \item ConclusiĂ³n: \textbf{Adam con lr $= 10^{-3}$} como
