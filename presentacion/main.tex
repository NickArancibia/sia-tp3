\begin{frame}{Arquitectura del MLP (Clase 11, Clase 13 - Backprop)}
  \textbf{Forward}: $V_j^m = \theta\!\left(\sum_k V_k^{m-1} \cdot w_{jk}^m\right)$
  con $V^0 = x$, $O = V^M$.

  \textbf{Loss (MSE)}: $E(w) = \frac{1}{2N} \sum_{\mu=1}^N \sum_i \big(\zeta_i^\mu - O_i^\mu\big)^2$

  \textbf{Backward (regla de la cadena)}:
  \[
    \delta^{(L)} = (O - t) \odot f'(h^{(L)}), \quad
    \delta^{(m)} = \big(W^{(m+1)\top}\delta^{(m+1)}\big) \odot f'(h^{(m)})
  \]
  \[
    \frac{\partial E}{\partial W^{(m)}} = \frac{1}{N}\,\delta^{(m)}\,V^{(m-1)\top}, \quad
    w^{\text{nuevo}} = w^{\text{ant}} - \eta\,\frac{\partial E}{\partial w}
  \]

  \textbf{Capacidad y generalización (Clase 13 - Regularización)}:
  más capacidad $\Rightarrow$ más riesgo de overfitting; brecha
  train$-$validación crece con la capacidad.
\end{frame}

\section{EJ1 --- Prevención de fraude}

\begin{frame}{EJ1 --- Contexto y setup}
  \small
  \textbf{Problema}: CompanyX usa un modelo grande y caro, \textbf{BigModel},
  para estimar la probabilidad de fraude en transacciones online.

  \vspace{0.4em}
  \textbf{Objetivo}: construir un modelo más chico, \textbf{TinyModel}, que
  aproxime la salida de BigModel con menor costo de inferencia.

  \vspace{0.4em}
  \textbf{Qué se pide estudiar:}
  \begin{itemize}
    \item Comparar un perceptrón simple lineal contra uno no lineal.
    \item Elegir el modelo con mejor capacidad de aprendizaje para generalización.
    \item Recomendar un umbral de detección de fraude.
  \end{itemize}

  \vspace{0.3em}
  \textbf{Criterio de negocio}: priorizar la reducción de falsos negativos,
  porque un fraude real no detectado es más costoso que revisar una operación
  sospechosa.
\end{frame}

\begin{frame}{EJ1 --- Normalización con z-score}
  \small
  Usamos \textbf{z-score} para llevar las variables a una escala comparable
  antes del entrenamiento.

  \vspace{0.4em}
  \[
    z = \frac{x - \mu}{\sigma}
  \]

  \scriptsize
  donde $\mu$ es la \textbf{media} y $\sigma$ es el \textbf{desvío estándar}.

  \vspace{0.4em}
  \scriptsize
  Ejemplo real sobre \texttt{amount\_usd} del dataset:

  \[
    x =
    \begin{bmatrix}
      42.07
    \end{bmatrix},
    \qquad
    \mu =
    \begin{bmatrix}
      109.07
    \end{bmatrix},
    \qquad
    \sigma =
    \begin{bmatrix}
      157.74
    \end{bmatrix}
  \]

  \[
    z =
    \begin{bmatrix}
      \frac{42.07 - 109.07}{157.74}
    \end{bmatrix}
    =
    \begin{bmatrix}
      -0.425
    \end{bmatrix}
  \]
\end{frame}

\begin{frame}{EJ1 --- Métricas de evaluación}
  \small
  Como criterio de diseño, queremos \textbf{minimizar los falsos negativos (FN)}:
  casos donde una operación fraudulenta se clasifica como no fraudulenta.
  Aun así, también contemplamos el resto de los errores.

  \vspace{0.4em}
  \begin{itemize}
    \item \textbf{F2}
    \[
      F_\beta = (1 + \beta^2) \cdot \frac{\mathrm{Precision}\,\mathrm{Recall}}{\beta^2\,\mathrm{Precision} + \mathrm{Recall}}
      \qquad \Rightarrow \qquad
      F_2 = 5 \cdot \frac{\mathrm{Precision}\,\mathrm{Recall}}{4\,\mathrm{Precision} + \mathrm{Recall}}
    \]

    \item \textbf{MSE}
    \[
      \mathrm{MSE} = \frac{1}{N} \sum_{i=1}^{N} \left(y_i - \hat{y}_i\right)^2
    \]
    \scriptsize donde $y_i$ es la salida de \textbf{BigModel} y $\hat{y}_i$ es la predicción de nuestro modelo.

    \item \textbf{Costo = $2\,FN + FP$}
  \end{itemize}
\end{frame}

\section{EJ2 — Clasificación de dígitos}

\begin{frame}{Contexto del ejercicio}
  \textbf{Problema}: CompanyX necesita un sistema automático para clasificar
  dígitos escritos a mano (0--9) a partir de imágenes de $28\times28$ pixels.

  \textbf{Modelo}: Perceptrón Multicapa (MLP) con salida de 10 neuronas
  (una por dígito), entrenado con MSE.

  \textbf{Estrategia experimental}:
  \begin{enumerate}
    \item Ver dependencia entre batch size y learning rate
    \item Ver variación de accuracy según optimizador y learning rate
    \item Explorar arquitecturas: ¿más capas o más nodos por capa?
    \item Evaluar funciones de activación en capas ocultas
    \item Analizar el mejor modelo sobre el conjunto de test
  \end{enumerate}

  \textbf{División}: 80\% train / 20\% val (estratificado) sobre
  \texttt{digits.csv}; \texttt{digits\_test.csv} como ``mundo real''.
\end{frame}

\begin{frame}{Exp2 — ¿Por qué Adam con lr $= 10^{-3}$?}
  \begin{itemize}
    \item \textbf{Adam} adapta la tasa de aprendizaje por parámetro
      (momentos de primer y segundo orden), por lo que trabaja bien
      con lr bajos y converge rápido.
    \item \textbf{GD} no tiene adaptación: necesita lr alto para
      moverse, pero eso genera inestabilidad.
    \item \textbf{Momentum} queda en un punto intermedio: acelera
      en direcciones consistentes pero no adapta por parámetro.
    \item Conclusión: \textbf{Adam con lr $= 10^{-3}$} como
      optimizador base para los experimentos siguientes.
  \end{itemize}
\end{frame}

\begin{frame}{Exp3 — ¿Más capas o más nodos por capa?}
  \textbf{Pregunta}: dada una arquitectura de hasta 2 capas ocultas,
  ¿cómo conviene distribuir los nodos entre la primera y la segunda
  capa? ¿Conviene una sola capa ancha o repartir en dos capas?

  \vspace{0.5em}
  \textbf{Diseño}: barrido 2D de \textbf{$n_1 \times n_2$} con
  $n_1, n_2 \in \{0, 8, 16, 32, 64, 128, 256, 512\}$.
  Convención: $n = 0$ significa que esa capa \emph{no existe}
  (los pares $(0, n)$ y $(n, 0)$ son el mismo MLP $[784, n, 10]$, así
  que se computan una sola vez).

  \begin{itemize}
    \item Total: \textbf{57 configuraciones únicas} $\times$ 5 seeds = 285 runs.
    \item Tres clases de arquitectura cubiertas en una sola grilla:
      \emph{regresión logística} (sin capas ocultas), \emph{1 capa
      oculta} de tamaño variable, y \emph{2 capas ocultas} con todas
      las combinaciones.
  \end{itemize}

  \vspace{0.5em}
  \textbf{Configuración}: Adam lr$=10^{-3}$, batch=32, tanh,
  patience=50, max\_epochs=500, init scale=0.1.
\end{frame}

\begin{frame}{Exp3 — Heatmap de val accuracy ($n_1 \times n_2$, ReLU)}
  \fig{0.55}{../ej2/results/part2/two_layer_heatmap_relu/val_acc_heatmap.png}
  \scriptsize 5 seeds por celda. Cada cell: mean $\pm$ std.
  $n=0$ significa que esa capa no existe; los pares $(0, n)$ y $(n, 0)$
  son el mismo MLP $[784, n, 10]$ y se computan una sola vez.
\end{frame}

\begin{frame}{Exp3 — Lectura del heatmap}
  \begin{itemize}
    \item Selección de hiperparámetros realizada sobre \textbf{val\_acc}
      (no test) para evitar \emph{test set leakage}.
    \item La zona top del barrido está prácticamente empatada en val
      (diferencias de pocos milipuntos): la fila $n_1{=}0$ y la columna
      $n_2{=}0$ junto a las celdas grandes muestran que con ReLU una
      sola capa oculta de 256–512 neuronas alcanza el techo.
    \item \textbf{Wider domina sobre deeper}: agregar una segunda
      capa NO mejora consistentemente. La primera capa es la más
      crítica porque procesa los 784 píxeles crudos.
    \item \textbf{Decisión final}: para el modelo del equipo se eligió
      $[784, 512, 10]$ + ReLU (1 sola capa oculta) — está dentro del
      empate técnico de los top y es la arquitectura más simple,
      menos propensa a overfit.
    \item El test se evalúa una sola vez al final sobre el modelo ya
      elegido (ver Exp5).
  \end{itemize}
\end{frame}

\subsection{EJ3 — Camino al 98\%}

\begin{frame}{Cómo llegamos al 98\% — overview}
  El baseline EJ3 (arquitectura grande $[784,256,128,10]$ + tanh)
  llega a $\sim$95.5\%. Para superar el 98\% lo que faltaba era:
  \begin{enumerate}\setlength\itemsep{0.2em}
    \item \textbf{Activación ReLU} (en vez de tanh) — convergencia más
      rápida y sin saturación.
    \item \textbf{Arquitectura simple}: $[784, 512, 10]$ (1 sola capa
      oculta), consistente con el ganador del heatmap ReLU del Exp3.
    \item \textbf{Online augmentation}: por cada época y cada batch,
      aplicar transformaciones aleatorias a las imágenes (rotación,
      traslación). El modelo nunca ve dos veces el mismo input exacto.
    \item \textbf{Weight decay} ($\lambda = 10^{-4}$) y
      \textbf{learning rate más conservador} (Adam $\eta = 2\!\cdot\!10^{-4}$)
      para evitar overfitting con un train chico.
  \end{enumerate}
  \vspace{0.4em}
  \textbf{Sin necesidad de}: feature engineering manual (HOG, momentos
  de Hu, etc.), regularización por noise, dropout o ensemble.
  Demuestra que la elección correcta de arquitectura + augmentation
  puede llevar a un MLP simple a 98\%.
\end{frame}

\begin{frame}{Online augmentation — la pieza clave}
  \textbf{Idea}: en cada época y cada batch, transformar las imágenes
  del train con perturbaciones aleatorias \emph{antes} de pasarlas al
  modelo. El modelo ``ve'' versiones distintas del mismo dígito en
  cada paso de entrenamiento.
  \vspace{0.4em}

  \textbf{Transformaciones que usamos}:
  \begin{itemize}\setlength\itemsep{0.1em}
    \item \textbf{Rotación} aleatoria en un rango pequeño.
    \item \textbf{Traslación} aleatoria de pocos píxeles.
  \end{itemize}
  \vspace{0.4em}

  \textbf{Por qué funciona}:
  \begin{itemize}\setlength\itemsep{0.1em}
    \item Reduce el overfitting: el dataset efectivo es muchísimo
      más grande.
    \item Hace al modelo \emph{invariante} a esos cambios pequeños.
    \item Ayuda especialmente a clases minoritarias (5, 8): cada época
      ven nuevas variantes de las mismas pocas muestras.
  \end{itemize}
\end{frame}

\section{Conclusiones}

\begin{frame}{Conclusiones EJ2}
  \begin{itemize}
    \item \textbf{Exp1 (batch $\times$ lr)}: el learning rate óptimo depende
      del batch size. Online logra la mejor accuracy pero es 10$\times$ más
      lento. Se elige batch=32 como compromiso eficiencia/rendimiento.
    \item \textbf{Exp2 (optimizadores)}: cada optimizador tiene su rango
      óptimo de lr. Adam con lr=$10^{-3}$ domina en accuracy y velocidad.
    \item \textbf{Exp3 (arquitecturas)}: heatmap 2D ($n_1 \times n_2$).
      \emph{Wider domina sobre deeper} en este rango: la primera capa es
      la más crítica (procesa los 784 píxeles crudos), la segunda aporta
      poco. Ganador: $[784, 512, 512, 10]$.
    \item \textbf{Exp4 (activación)}: ReLU, tanh y logistic dan resultados
      aceptables. ReLU es preferida por convergencia más rápida y sin
      saturación en valores positivos.
    \item \textbf{Exp5 (mejor modelo)}: \textbf{[784, 512, 10]} (1 capa oculta) + ReLU + Adam
      lr=$10^{-3}$, batch=32. Ganador del heatmap ReLU.
    \item \textbf{Techo de test accuracy $\approx$ 90\%} por la ausencia
      del dígito 8 en train. El 8 se confunde con 5 y 3; el dígito 5
      (271 muestras) también sufre.
  \end{itemize}
\end{frame}
